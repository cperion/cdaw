-- ================================================================
-- PHASE 5: KERNEL
-- Zero sum types. Terra only.
-- ================================================================
module Kernel {

    -- Design note:
    --   • Kernel is the fully monomorphic execution surface.
    --   • Partial implementation remains acceptable as long as emitted entry
    --     points are valid and local jobs/devices degrade to stable fallback
    --     behavior (silence, passthrough, zero control) rather than malformed
    --     machine code or missing API functions.
    --   • The atomic reusable compile product is Kernel.Unit: one function,
    --     one owned state ABI.

    Unit = (
        TerraFunc fn,
        TerraType state_t
    ) unique

    Buffers = (
        TerraType mono_t,
        TerraType stereo_t,
        TerraType event_t,
        TerraType bus_array_t,
        TerraType state_t
    )

    State = (
        TerraType transport_t,
        TerraType control_t,
        TerraType dsp_t,
        TerraType launcher_t,
        TerraType voice_t,
        TerraType render_t
    )

    API = (
        TerraFunc init_fn,
        TerraFunc destroy_fn,
        TerraFunc render_block_fn,
        TerraFunc set_param_fn,
        TerraFunc queue_launch_fn,
        TerraFunc queue_scene_fn,
        TerraFunc stop_track_fn,
        TerraFunc note_on_fn,
        TerraFunc note_off_fn,
        TerraFunc poly_pressure_fn,
        TerraFunc cc_fn,
        TerraFunc pitch_bend_fn,
        TerraFunc timbre_fn,
        TerraFunc get_peak_fn,
        TerraFunc get_position_fn,
        TerraFunc get_param_fn,
        TerraFunc get_mod_fn
    )

    Project = (
        Kernel.Buffers buffers,
        Kernel.State   state,
        Kernel.API     api,
        TerraFunc      entry
    ) unique
    methods {
        entry_fn() -> TerraFunc
    }
}
